		   System.out.println("Your total fees are="+ fee);
		}
		else if((hours>3) && (hours<23))
		{
			double fee2;
			fee2=fee+(hours-3)*0.5;
		    System.out.println("Your total fees are=" +fee2);
		
		
		}
		else if(hours==24)
		{
			double fee3=10;
			System.out.println("Your total fees (24hrs parking) are=" +fee3);
		}
			
		return ;

		
		
	}

}
